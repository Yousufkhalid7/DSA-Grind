11. Container With Most Water
Approach: Two Pointers

Solution:
class Solution(object):
    def maxArea(self, height):
        max_area = 0
        i = 0
        j = len(height) - 1
        while i < j:
            area = min(height[i], height[j])*(j-i)
            max_area = max(area, max_area)
            if height[i] < height[j]:
                i += 1
            else:
                j -= 1
        return max_area
