530. Minimum Absolute Difference in BST
Approach: Inorder BST

Solution:
class Solution(object):
    def getMinimumDifference(self, root):
        if root is None:
            return 0
        inorder = []
        def calc(node):
            if node is None:
                return
            calc(node.left)
            inorder.append(node.val)
            calc(node.right)
        calc(root)
        diff = float('inf')
        for i in range(1, len(inorder)):
            diff = min(diff, inorder[i] - inorder[i-1])
        return diff
