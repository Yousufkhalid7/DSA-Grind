637. Average of Levels in Binary Tree
Approach: Deque with Breadth First Search

Solution:
from collections import deque
class Solution(object):
    def averageOfLevels(self, root):
        result = []
        if root is None:
            return result
        queue = deque([root])
        while queue:
            level = len(queue)
            level_sum = 0
            
            for i in range(level):
                node = queue.popleft()
                level_sum += node.val

                if node.left:
                    queue.append(node.left)
                if node.right:
                    queue.append(node.right)

            result.append(float(level_sum)/level)
        return result
