128. Longest Consecutive Sequence
Approach: Sliding Window

Solution:
class Solution(object):
    def longestConsecutive(self, nums):
        if not nums:
            return 0
        nums = list(set(nums))
        nums.sort()
        count = 1
        max_length = 1
        for i in range(1, len(nums)):
            if nums[i] - nums[i-1] == 1:
                count += 1
                max_length = max(max_length, count)
            else:
                count = 1
        return max_length
