39. Combination Sum
Approach: Backtracing

Solution:
class Solution(object):
    def combinationSum(self, candidates, target):
        current = []
        result = []
        def dfs(index, total):
            if total > target:
                return
            if total == target:
                result.append(current[:])
                return
            for i in range(index, len(candidates)):
                current.append(candidates[i])
                dfs(i, total + candidates[i])
                current.pop()
        dfs(0, 0)
        return result
